function escHTML(text) {
    return String(text)
        .replace(/&/g, '&amp;')
        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;');
}

async function loadProxies() {
    try {

        const res = await axios.get(PROXY_LIST_URL, {
            timeout: 15000
        });

        proxies = res.data
            .split('\n')
            .map(p => p.trim())
            .filter(p => p.includes('://'));

        console.log(`${E.ok} Loaded ${proxies.length} proxies`);

    } catch (err) {

        console.log(
            `${E.no} Gagal load proxy: ${err.message}`
        );
    }
}

function getRandomProxy() {

    if (proxies.length === 0) return null;

    return proxies[
        Math.floor(Math.random() * proxies.length)
    ];
}

function getProxyIp(proxyUrl) {

    try {

        const u = new URL(proxyUrl);
        return u.hostname;

    } catch {

        return proxyUrl;
    }
}

async function checkProxy(proxyUrl) {

    try {

        const agent = new HttpsProxyAgent(proxyUrl);

        const res = await axios.get(
            'https://api.ipify.org?format=json',
            {
                httpsAgent: agent,
                httpAgent: agent,
                timeout: 10000
            }
        );

        if (res.data?.ip) {

            return {
                active: true,
                ip: res.data.ip,
                agent
            };
        }

        return {
            active: false
        };

    } catch {

        return {
            active: false
        };
    }
}

function getPayload(session) {

    return {
        plan: 'plus',
        payment: 'shortlink',
        currency: 'Indonesia',
        session: session
    };
}

function parseErrorMessage(msg) {

    if (!msg)
        return 'Unknown error';

    if (
        msg.includes(
            'Could not parse your authentication token'
        )
    ) {

        return `${E.red}session token kadaluwarsa, ambil lagi disini:\nhttps://chatgpt.com/api/auth/session`;
    }

    if (
        msg.includes('Could not parse')
    ) {

        return `${E.red}session token kadaluwarsa, ambil lagi disini:\nhttps://chatgpt.com/api/auth/session`;
    }

    if (
        msg.includes('Invalid access token')
    ) {

        return `${E.red}token gak valid\nsession salah atau sudah expired`;
    }

    return msg;
}

        );

    for (let i = 0; i < 5; i++) {

        const proxyUrl =
            getRandomProxy();

        if (!proxyUrl) {

            lastError =
                'Proxy kosong';

            break;
        }

        console.log(
            `${E.reload} Checking proxy: ${proxyUrl}`
        );

        const proxyCheck =
            await checkProxy(proxyUrl);

        if (!proxyCheck.active) {

            console.log(
                `Proxy mati: ${getProxyIp(proxyUrl)}`
            );

            continue;
        }

        await ctx.reply(
            `${E.globe} OK, dapat ip: ${proxyCheck.ip}`
        );

        try {

            const response =
                await axios.post(
                    PAYMENT_URL,
                    getPayload(session),
                    {
                        httpsAgent:
                            proxyCheck.agent,

                        httpAgent:
                            proxyCheck.agent,

                        headers:
                            getHeaders(true),

                        timeout: 20000
                    }
                );

            if (
                response.data?.success &&
                response.data?.url
            ) {

                await ctx.reply(
                    `${E.ok} <b>OK, berhasil ambil link payment:</b>`,
                    {
                        parse_mode:
                            'HTML'
                    }
                );

                return ctx.reply(
                    `${E.link} ${response.data.url}`
                );
            }

            const msg =
                response.data?.msg ||
                response.data?.message ||
                'Unknown';

            const reply =
                parseErrorMessage(msg);

            return ctx.reply(
                `${E.no} ${escHTML(reply)}`,
                {
                    parse_mode:
                        'HTML'
                }
            );

        } catch (err) {

            if (err.response) {

                const status =
                    err.response.status;

                const data =
                    err.response.data || {};

                const msg =
                    data.msg ||
                    data.message ||
                    JSON.stringify(data);

                console.log(
                    'STATUS:',
                    status
                );

                console.log(
                    'DATA:',
                    data
                );

                if (
                    msg.includes(
                        'Could not parse'
                    ) ||
                    msg.includes(
                        'Invalid access token'
                    )
                ) {

                    const reply =
                        parseErrorMessage(msg);

                    return ctx.reply(
                        `${E.no} ${escHTML(reply)}`,
                        {
                            parse_mode:
                                'HTML'
                        }
                    );
                }

                lastError =
                    `HTTP ${status} | ${msg}`;

            } else {

                lastError =
                    err.code ||
                    err.message;
            }

            console.log(
                `Attempt ${i + 1} failed: ${lastError}`
            );
        }
    }

    await ctx.reply(
        `${E.warn} Semua proxy gagal, mencoba tanpa proxy...`
    );

    try {

        const response =
            await axios.post(
                PAYMENT_URL,
                getPayload(session),
                {
                    headers:
                        getHeaders(false),

                    timeout: 20000
                }
            );

        if (
            response.data?.success &&
            response.data?.url
        ) {

            await ctx.reply(
                `${E.ok} <b>Berhasil tanpa proxy!</b>`,
                {
                    parse_mode:
                        'HTML'
                }
            );

            return ctx.reply(
                `${E.link} ${response.data.url}`
            );
        }

        const msg =
            response.data?.msg ||
            response.data?.message ||
            'Gagal tanpa proxy juga';

        const reply =
            parseErrorMessage(msg);

        return ctx.reply(
            `${E.no} ${escHTML(reply)}`,
            {
                parse_mode:
                    'HTML'
            }
        );

    } catch (err) {

        let finalError = '';

        if (err.response) {

            const status =
                err.response.status;

            const data =
                err.response.data || {};

            const msg =
                data.msg ||
                data.message ||
                JSON.stringify(data);

                'STATUS:',

            console.log(
                'DATA:',
                data
            );

            finalError =
                `HTTP ${status} | ${msg}`;

                err.code ||
                err.message ||
                lastError;
        }

        return ctx.reply(
            `${E.no} Semua cara gagal.\n${escHTML(finalError)}`,

                parse_mode:

        $create = curlReq(
            "https://www.dreamfaceapp.com/dw-server/face/animate_image",
            $data,
            $saveHeaders,
            $proxy
        );

        file_put_contents("enhance.txt", $create);

        $animate_id = get_between($create, '"animate_image_id":"', '","');

        if(!$animate_id){
            editMessage($chat_id, $message_id, "❌ Enhance gagal dimulai");
            deleteFromImgBB($delete_url);
            @unlink($lock);
            return;
        }

        $start = time();

        while(true){

            $pollData = '{
                "user_id":"'.$user_id.'",
                "account_id":"'.$account_id.'",
                "animate_id_list":["'.$animate_id.'"],
                "platform_type":"MOBILE",
                "tenant_name":"dream_face",
                "platformType":"MOBILE",
                "tenantName":"dream_face"
            }';

            $pollRes = curlReq(
                "https://www.dreamfaceapp.com/dw-server/face/animate_image_list_poll",
                $pollData,
                $saveHeaders,
                $proxy
            );

            $state  = get_between($pollRes, '"state":"', '"');
            $result = get_between($pollRes, '"picture_path_list":["', '"');

            if($state == "finish" && $result){

                editMessage($chat_id, $message_id, "✅ OK, foto siap dikirim");

                sendPhoto($chat_id, $result, "✨ HD FOTO");

                deleteFromImgBB($delete_url);

                @unlink($lock);

                return;
            }

            if($state == "fail"){
                editMessage($chat_id, $message_id, "❌ Enhance gagal");
                deleteFromImgBB($delete_url);
                @unlink($lock);
                return;
            }

            if(time() - $start > 1800){
                editMessage($chat_id, $message_id, "❌ Enhance timeout");
                deleteFromImgBB($delete_url);
                @unlink($lock);
                return;
            }

            sleep(5);
        }

    } catch(Exception $e){

        sendMessage($chat_id, "❌ Bot Error");

        @unlink($lock);
    }
}

function uploadToImgBB($file_id)
{
    global $api, $botToken, $imgbbKey;

    $res  = @file_get_contents($api."/getFile?file_id=".urlencode($file_id));
    $data = json_decode($res, true);

    $file_path = $data['result']['file_path'] ?? null;

    if(!$file_path){
        return null;
    }

    $file_url = "https://api.telegram.org/file/bot".$botToken."/".$file_path;

    $ch = curl_init();

    curl_setopt($ch, CURLOPT_URL, "https://api.imgbb.com/1/upload?key=".urlencode($imgbbKey));
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_POST, true);
    curl_setopt($ch, CURLOPT_POSTFIELDS, http_build_query(['image' => $file_url]));
    curl_setopt($ch, CURLOPT_TIMEOUT, 60);
    curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);

    $res  = curl_exec($ch);
    curl_close($ch);

    $json = json_decode($res, true);

    if(!empty($json['data']['url'])){
        return [
            'url'        => $json['data']['url'],
            'delete_url' => $json['data']['delete_url'] ?? null
        ];
    }

    $image_data = @file_get_contents($file_url);

    if(!$image_data){
        return null;
    }

    $ch = curl_init();

    curl_setopt($ch, CURLOPT_URL, "https://api.imgbb.com/1/upload?key=".urlencode($imgbbKey));
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_POST, true);
    curl_setopt($ch, CURLOPT_POSTFIELDS, http_build_query(['image' => base64_encode($image_data)]));
    curl_setopt($ch, CURLOPT_TIMEOUT, 60);
    curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);

    $res  = curl_exec($ch);
    curl_close($ch);

    $json = json_decode($res, true);

    if(!empty($json['data']['url'])){
        return [
            'url'        => $json['data']['url'],
            'delete_url' => $json['data']['delete_url'] ?? null
        ];
    }

    return null;
}

function bot($method, $data = [])
{

function curlReq($url, $post = null, $headers = [], $proxy = null)
{
    $ch = curl_init();

    curl_setopt($ch, CURLOPT_URL, $url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
    curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
    curl_setopt($ch, CURLOPT_SSL_VERIFYHOST, false);
    curl_setopt($ch, CURLOPT_CONNECTTIMEOUT, 30);
    curl_setopt($ch, CURLOPT_TIMEOUT, 60);

    if($post){
        curl_setopt($ch, CURLOPT_POST, true);
        curl_setopt($ch, CURLOPT_POSTFIELDS, $post);
    }

    if($headers){
        curl_setopt($ch, CURLOPT_HTTPHEADER, $headers);
    }

    if($proxy){
        curl_setopt($ch, CURLOPT_PROXY, $proxy);
        curl_setopt($ch, CURLOPT_HTTPPROXYTUNNEL, true);
    }

    $res = curl_exec($ch);

    curl_close($ch);

    return $res;
}

function get_between($string, $start, $end)
{
    $string = " ".$string;
    $ini    = strpos($string, $start);

    if($ini === false){
        return "";
    }

    $ini += strlen($start);
    $len  = strpos($string, $end, $ini);

    if($len === false){
        return "";
    }

    return substr($string, $ini, $len - $ini);
}

function generateRandomString($length = 10)
{
    $characters       = 'abcdef0123456789';
    $charactersLength = strlen($characters);
    $randomString     = '';

    for($i = 0; $i < $length; $i++){
        $randomString .= $characters[rand(0, $charactersLength - 1)];
    }

    return $randomString;
}

   # Buat folder kerja

   sudo apt-get update -qq

B. Command Run Bot:

   # Jalankan gpt.js (Node.js bot)
   cd /home/ubuntu/bots && node gpt.js &

   # Jalankan bot.php (PHP bot)
   cd /home/ubuntu/bots && php bot.php &

C. Command Restart Jika Error (gunakan script run_bots.sh):

   #!/bin/bash
   # Script: /home/ubuntu/bots/run_bots.sh
   cd /home/ubuntu/bots

   while true; do
       echo "[$(date)] Starting gpt.js..."
       node gpt.js &
       GPT_PID=$!

       echo "[$(date)] Starting bot.php..."
       php bot.php &
       PHP_PID=$!

       echo "[$(date)] Bots running: gpt.js=$GPT_PID, bot.php=$PHP_PID"

       # Tunggu salah satu proses mati
       wait -n $GPT_PID $PHP_PID 2>/dev/null

       echo "[$(date)] Bot crashed! Restarting in 5s..."

       # Kill semua proses bot yang masih jalan
       kill $GPT_PID $PHP_PID 2>/dev/null
       wait $GPT_PID $PHP_PID 2>/dev/null

       sleep 5
   done

   # Jalankan script:
   chmod +x /home/ubuntu/bots/run_bots.sh
   bash /home/ubuntu/bots/run_bots.sh

D. Cara Cek Log:

   # Cek proses yang jalan
   ps aux | grep -E "gpt.js|bot.php" | grep -v grep

   # Jika pakai run_bots.sh di background:
   # Redirect output ke log file:
   bash /home/ubuntu/bots/run_bots.sh > /home/ubuntu/bots/bot.log 2>&1 &

   # Cek log:
   tail -f /home/ubuntu/bots/bot.log

================================================================================

- Shell sempat error "connection unavailable" saat session restart
  Solusi: Retry/wait sampai shell pulih, biasanya 5-10 detik

- bot.php kadang menampilkan "[ ! ] Telegram Connection Error"
  Solusi: Ini normal, bot akan auto-retry setiap 3 detik

- gpt.js menampilkan deprecation warning tentang punycode module
  Solusi: Bisa diabaikan, tidak mempengaruhi fungsi bot

- gpt.js menampilkan ExperimentalWarning tentang CommonJS loading ES Module
  Solusi: Bisa diabaikan, tidak mempengaruhi fungsi bot
